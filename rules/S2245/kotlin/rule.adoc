== Sensitive Code Example

----
val random = Random() // Noncompliant: Random() is not a secure random number generaotr
val bytes = ByteArray(20)
random.nextBytes(bytes)
----

== Compliant Solution

----
val random = SecureRandom() // Compliant
val bytes = ByteArray(20)
random.nextBytes(bytes)
----

== Noncompliant Code Example

----
val x = (y / 2 + 1)  // Compliant even if the parentheses are ignored by the compiler

if (a && ((x + y > 0))) {  // Noncompliant
  //...
}

return ((x + 1))  // Noncompliant
----

== Compliant Solution

----
val x = (y / 2 + 1)

if (a && (x + y > 0)) {
  //...
}

return (x + 1)
----

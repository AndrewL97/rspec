== Noncompliant Code Example

----
public class MyClass<TYPE> { // Noncompliant
  <TYPE> void method(TYPE t) { // Noncompliant
  }
}
----

== Compliant Solution

----
public class MyClass<T> { 
  <T> void method(T t) {
  }
}
----
